两个函数
- copy_to_user
- copy_from_user
```c
static ssize_t m_dev_read(struct file *file, char __user *buf, size_t size, loff_t *off){
    printk("this is read\n");
    struct m_dev *dev_test =(struct m_dev *)file->private_data;
    // 第一个是to 第二个是from 第三个是大小
    if(copy_to_user(buf,dev_test->kbuf,strlen(dev_test->kbuf))<0){
        printk("read error\n");
        return -1;
    }
    return 0;
}
static ssize_t m_dev_write(struct file *file, const char __user *buf, size_t size, loff_t *off){
    printk("this is write\n");
    struct m_dev *dev_test =(struct m_dev *)file->private_data;
    // 第一个是to 第二个是from 第三个是大小
    if(copy_from_user(dev_test->kbuf ,buf ,size)<0){
        printk("write error\n");
        return -1;
    }
    return 0;
}
```
